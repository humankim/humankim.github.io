### 미로 탐색 문제풀이
#### 문제 : https://www.acmicpc.net/problem/2178

- 이동 할 수 있는 최소 칸 수(가장 빠른 길) -> BFS
1.  BFS로 탐색 0, 0부터 시작하여 거리를 누적시키고 마지막 좌표의 거리를 출력
2. BFS는 단계별로 진행된다는 사실을 이용

```C++
void bfs(int x, int y) {
    queue<pair<int, int>> q;
    q.push(make_pair(x, y));
    check[x][y] = true;
    dist[x][y] = 1;

    while(!q.empty()) {
        x = q.front().first;
        y = q.front().second;
        q.pop();
        for(int i=0; i<4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(nx >= 0 && nx < N && ny >= 0 && ny < M) {
                if(a[nx][ny]==1 && check[nx][ny] == false) {
                    q.push(make_pair(nx, ny));
                    dist[nx][ny] = dist[x][y] + 1; //거리 증가
                    check[nx][ny] = true; // 방문 체크
                }
            }
        }
    }
}

int main()
{
    cin >> N >> M;
    for(int i=0; i < N; i++) {
        for(int j=0; j < M; j++) {
            scanf("%1d", &a[i][j]);
        }
    }
    bfs(0, 0); // 첫 좌표부터 시작

    cout << dist[N-1][M-1] << '\n'; //마지막 좌표의 거리 출력
    return 0;
}
```
