### 다리 만들기 저지 풀이
#### 문제 : https://www.acmicpc.net/problem/2146

- 두 섬 연결하는 가장 짧은 다리 -> BFS
- 섬 그룹 나눠서 각각 그룹에서 다른 섬까지 거리 -> 각 그룹 BFS는 느릴 듯
- BFS로 그룹번호를 확장시킨 후, 서로 그룹번호가 다른 두 칸이 인접 -> 다리 고려
- 각각의 점을 시작점에 넣은 후, 거리를 구하는 방식

```C++
int a[100][100]; // 정점 번호 0 or 1
int g[100][100]; // 각 정점의 그룹 번호 
int d[100][100]; 
int dx[] = {0, 0, 1, -1};
int dy[] = {1, -1, 0, 0};
int cnt = 0;
void go1()
{
    //그룹핑
    for(int i=0; i<N; i++) {
        for(int j=0; j<N; j++) {
            if(a[i][j] == 1 && g[i][j] == 0) { // 정점번호 1이고 그룹번호 부여 안됐으면
                g[i][j] = ++cnt;
                queue<pair<int, int>> q;
                q.push(make_pair(i,j));
                while(!q.empty()) {
                    int x = q.front().first;
                    int y = q.front().second;
                    q.pop();
                    for(int k=0; k<4; k++) {
                        int nx = x + dx[k];
                        int ny = y + dy[k];
                        //같은 그룹끼리  묶기 
                        if(nx >= 0 && nx < N && ny >= 0 && ny < N) {
                          if(a[nx][ny] == 1 && g[nx][ny] == 0) {
                              g[nx][ny] = cnt;
                              q.push(make_pair(nx, ny));
                          }
                        }
                    }
                }
            }
        }
    }
    
}

void go2_long()
{
    int ans = -1;
    //L번섬에서의 거리를 Q에 넣어줌 
    for(int l = 1; l<= cnt; l++) {
        queue<pair<int, int>> q;
        for(int i=0; i<N; i++) {
            for(int j=0; j<N; j++) {
                d[i][j] = -1;  //거리 초기화
                if(g[i][j] == l) {  //L번 섬끼리 큐에 넣고 거리 0으로 한다. 즉, 시작점을 구했음
                    q.push(make_pair(i, j));
                    d[i][j] = 0;
                }
            }
        }
    }

    //시작점을 알았으니 거리를 다 구해보자
    while(!q.empty()) {
        int x = q.front().first;
        int y = q.front().second;
        q.pop();
        for(int k=0; k<4; k++) {
            int nx = x + dx[k];
            int ny = y + dy[k];
            //거리를 구하자 
            if(nx >= 0 && nx < N && ny >= 0 && ny < N) {
                if(d[nx][ny] == -1) {
                    d[nx][ny] = d[x][y] + 1;
                    q.push(make_pair(nx, ny));
                }
            }
        }
    }

    //섬이면서 그룹번호가 L번이 아닌 것
    for(int i=0; i<N; i++) {
        for(int j=0; j<N; j++) {
            if(a[i][j] == 1 && g[i][j] != l) {
                if(ans == -1 || d[i][j]-1 < ans) {
                    ans = d[i][j] - 1;
                }
            }
        }
    }

    cout << ans << '\n'; 
}

void go2_short()
{
    int ans = -1;
    // 모든 섬 거리를 Q에 넣어줌 
    queue<pair<int, int>> q;
    for(int i=0; i<N; i++) {
        for(int j=0; j<N; j++) {
            d[i][j] = -1;  //거리 초기화
            if(a[i][j] == 1) {
                q.push(make_pair(i, j));
                d[i][j] = 0;
            }
        }

    //시작점을 알았으니 거리를 다 구해보자
    while(!q.empty()) {
        int x = q.front().first;
        int y = q.front().second;
        q.pop();
        for(int k=0; k<4; k++) {
            int nx = x + dx[k];
            int ny = y + dy[k];
            //거리를 구하자 
            if(nx >= 0 && nx < N && ny >= 0 && ny < N) {
                if(d[nx][ny] == -1) {
                    d[nx][ny] = d[x][y] + 1;
                    //이것이 go2_long에 비해 아래와 같이 추가된다
                    g[nx][ny] = g[x][y]; // 섬이 아니라도 섬이라 치고 그룹번호 정해줌
                    q.push(make_pair(nx, ny));
                }
            }
        }
    }

    // 인접한 (i,j) - (x,y) => dx, dy를 이용
    for(int i=0; i<N; i++) {
        for(int j=0; j<N; j++) {
            for(int k=0; k<N; k++) {
                int x = i + dx[k];
                int y = i + dy[k];
                if(x >= 0 && x < N && y >= 0 && y < N) {
                    if(g[i][j] != g[x][y]) { //그룹번호가 서로 다르면 선이 완성
                        if(ans == -1 || ans > d[i][j] + d[x][y]) {
                            ans = d[i][j] + d[x][y];
                        }
                    }
                }
            }
        }
    }

    cout << ans << '\n'; 
}




int main()
{
    cin >> N;
    for(int i=0; i<N; i++) {
        for(int j=0; j<N; j++) {
            scanf("%d", &a[i][j]);
        }
    }
    go1();
    //go2_long(); // 1초 가까이 소요 (BFS 여러번)
    go2_short(); // 0~4 ms 정도만 소요 (BFS 1번만으로 해결)
    return 0;
}
```
