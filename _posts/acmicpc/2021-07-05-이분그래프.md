### 이분 그래프 문제풀이
#### 문제 : https://www.acmicpc.net/problem/1707

- 정점과 간선의 정보로 그룹번호(1 or 2)를 기록
1.  DFS 돌리고 한번 돌릴때마다 개수를 증가시키면 된다


```C++

vector<int> a[1001];
bool color[1001];  //0:방문안함, 1:1번그룹, 2:2번그룹

//그룹핑만 해놓기 
void dfs1(int node, int c) {
    color[node] = c;

    for(int i=0; i<a[node].size(); i++) {
        int next = a[node][i];
        if(color[next]==0) {
            dfs1(next, 3-c); //3-c : 1을 2로, 2를 1로
        }
    }
}

//그룹핑하고 결과까지 한방에
bool dfs2(int node, int c) {
    color[node] = c;
    for(int i=0; i<a[node].size(); i++) {
        int next = a[node][i];
        if(color[next]==0)) {
            if (dfs2(next, 3-c)  == false) { //true면 loop 계속 
                return false;
            }
        } else if(color[next] == color[node]) {
            return false; //그룹번호가 같으면 false
        }
    }
    return true;
}

int main()
{
    cin >> N >> M;
    for(int i=0; i < M; i++) {
        cin >> from >> to;
        a[from].push_back(to);
        a[to].push_back(from);
    }
    bool ok = true;
    #if 0
    for(int i=1; i <= N; i++) {
        if(color[i] == 0) {
            dfs1(i, 1);
        }
    }
    for(int i=1; i <= N; i++) {
        for(int k=0; k < a[i].size(); k++) {
            int j = a[i][k];
            if(color[i] == color[j]) {
                ok = false;
            }
        }
    }
    #else
    for(int i=1; i <= N; i++) {
        if(color[i] == 0) {
            if(dfs2(i, 1) == false) {
                ok = false;
            }
        }
    }
    #endif
    cout << (ok?"YES":"NO") << '\n';
    return 0;
}
```
