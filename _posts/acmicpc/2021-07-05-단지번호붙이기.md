### 단지번호 붙이기 문제풀이
#### 문제 : https://www.acmicpc.net/problem/2667

- 단지별로 그룹핑하는 문제
1.  DFS/BFS 모두 가능
2.  호출부에서는 카운트를 증가시킴
3.  DFS/BFS 함수 내에서는 4방향으로 그룹핑을 하고 입력받은 카운트만을 호출
4.  그룹 카운트를 또 증가시킨다
5.  오름 차순 정렬 시,  0번째 인덱스는 0이 됨을 기억하고 1번째 인덱스부터 출력


```C++
int N;
int a[26][26];
int color[26][26];  //0:방문안함, 1~:단지번호
int dx[] = {0, 0, -1, 1};
int dy[] = {-1, 1, 0, 0};
int grpcnt[626];

//DFS를 좌표만큼 호출
void dfs(int x, int y, int c) {
    color[x][y] = c;
    grpcnt[c]++;

    for(int i=0; i<4; i++) {
        int nx = x + dx[i];
        int ny = y + dy[i];
        if(nx >=0 && nx < N && ny >= 0 && ny < N) {
            if(a[nx][ny]==1 && color[nx][ny]==0) {
                dfs(nx, ny, c);
            }
        }
    }

}

void bfs(int x, int y, int c) {
    queue<pair<int, int>> q;
    q.push(make_pair(x, y));
    color[x][y]  = c;
    grpcnt[c]++;

    while(!q.empty()) {
        x = q.front().first;
        y = q.front().second;
        q.pop();
        for(int i=0; i<4; i++) {
            int nx = x + dx[i];
            int ny = y + dy[i];
            if(nx >=0 && nx < N && ny >= 0 && ny < N) {
                if(a[nx][ny]==1 && color[nx][ny]==0) {
                    q.push(make_pair(nx, ny));
                    color[nx][ny] = c;
                    grpcnt[c]++;
                }
            }
        }
    }

}

int main()
{
    cin >> N;
    for(int i=0; i < N; i++) {
        for(int j=0; j < N; j++) {
            scanf("%1d", &a[i][j]);
        }
    }
    int cnt = 0;
    for(int i=0; i < N; i++) {
        for(int j=0; j < N; j++) {
            if(a[i][j] == 1 && color[i][j] == 0) {
                dfs(i, j, ++cnt);
            }
        }
    }
    sort(grpcnt, grpcnt+cnt);
    for(int i=1; i<=cnt; i++) {
        cout << grpcnt[i] << '\n';
    }
    return 0;
}
```
